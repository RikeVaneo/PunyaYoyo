DELIMITER $$

CREATE PROCEDURE listProductsBySupplier(
    IN pSupplierID INT,
    INOUT productList VARCHAR(5000)
)
BEGIN
    DECLARE done INT DEFAULT 0;
    DECLARE pName VARCHAR(100);
    DECLARE pStock INT;

    DECLARE prodCursor CURSOR FOR
        SELECT p.productName, ps.stock
        FROM products p
        JOIN product_supplies ps ON p.productCode = ps.productCode
        WHERE ps.supplierID = pSupplierID;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    OPEN prodCursor;

    read_loop: LOOP
        FETCH prodCursor INTO pName, pStock;
        IF done = 1 THEN 
            LEAVE read_loop;
        END IF;

        SET productList = CONCAT(productList, pName, ' - ', pStock, '; ');
    END LOOP;

    CLOSE prodCursor;

END $$

DELIMITER ;
